def merge_bodies(body1, body2): # Tách param=value của body1 và body2 thành list params1 = body1.split('&') params2 = body2.split('&') # Tạo set chứa tên param trong body1 keys1 = set(param.split('=')[0] for param in params1 if '=' in param) # Lưu phần kết quả merged = params1.copy() for param in params2: if '=' not in param: continue # bỏ qua nếu không hợp lệ key = param.split('=')[0] if key not in keys1: merged.append(param) return '&'.join(merged) # Test case body1 = "a=1&b=2&e=" body2 = "a=&cc=2&c=3" result = merge_bodies(body1, body2) print(result)
