function getTextHint(element) { let current = element; const form = element.closest("form") || document.body; let root = null; // Tìm thẻ cha cao nhất (nhưng vẫn thuộc form hoặc body) while (current && current !== form) { root = current; current = current.parentElement; } if (!root) return ""; const labelTexts = Array.from(root.querySelectorAll("label")) .map((node) => node.textContent?.trim()) .filter((text) => text && text.length > 0); const otherTexts = Array.from(root.querySelectorAll("span, p, th, div")) .map((node) => node.textContent?.trim()) .filter((text) => text && text.length > 0); const allTexts = [...labelTexts, ...otherTexts]; return allTexts.join(" ").trim(); } } } } }
