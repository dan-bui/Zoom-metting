import base64 def decode_base64_urlsafe(data): # Thêm padding nếu thiếu (độ dài phải chia hết cho 4) data += '=' * (-len(data) % 4) try: # Decode theo chuẩn base64 url-safe decoded_bytes = base64.urlsafe_b64decode(data) print("✅ Đã decode thành công!") print("🔹 Dạng byte (hex):", decoded_bytes.hex()) print("🔹 Dạng chuỗi (utf-8):") try: print(decoded_bytes.decode('utf-8')) except UnicodeDecodeError: print("(Không thể decode thành UTF-8, có thể là dữ liệu nhị phân)") except Exception as e: print("❌ Lỗi khi decode:", str(e)) # Ví dụ sử dụng if __name__ == "__main__": b64_input = input("Nhập chuỗi base64 url-safe cần decode: ").strip() decode_base64_urlsafe(b64_input)
