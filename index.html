const input = `aa aa bb bb bb c`; const lines = input.split('\n'); const result = []; let prev = null; let count = 0; for (let i = 0; i < lines.length; i++) { const current = lines[i]; const next = lines[i + 1]; if (current === prev) { count++; result.push(`${current}#${count}`); } else if (current === next) { count = 1; result.push(`${current}#${count}`); } else { result.push(current); // xuất hiện duy nhất count = 0; } prev = current; } console.log(result.join('\n'));
