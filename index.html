from urllib.parse import parse_qs def parse_body(body): parsed = parse_qs(body, keep_blank_values=True) return {k: v[0] for k, v in parsed.items()} def compare_bodies(body1, body2): d1 = parse_body(body1) d2 = parse_body(body2) if d1 == d2: return "trùng" elif all(k in d2 and d2[k] == v for k, v in d1.items()): return "1 là con của 2" elif all(k in d1 and d1[k] == v for k, v in d2.items()): return "2 là con của 1" else: return "2 body khác nhau (2 tập hợp độc lập)" # ===== Gán trực tiếp ===== body1 = "a=1&b=2" body2 = "a=1&b=2&c=3" # ===== So sánh và in kết quả ===== print(f"body1 = {body1}") print(f"body2 = {body2}") print(f"Kết luận: {compare_bodies(body1, body2)}")
