Get-ChildItem -Path "C:\Program Files" -Recurse -Directory -ErrorAction SilentlyContinue | Where-Object { (Get-Acl $_.FullName -ErrorAction SilentlyContinue).Access | Where-Object { $_.IdentityReference -match "BUILTIN\\Users" -and $_.FileSystemRights.ToString() -match "Write|Modify|FullControl" }} | Select-Object FullName
