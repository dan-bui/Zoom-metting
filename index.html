function getTextHint(element) { // Bước 1: xác định thẻ mẹ trực tiếp chứa input let current = element.parentElement; // Bước 2 và 3: tìm thẻ cha cao nhất chứa input bằng cách lặp từng cấp while (current && current !== document.body) { const siblings = Array.from(current.parentElement?.children || []); const hasSiblingWithInput = siblings.some((sibling) => { if (sibling === current) return false; return sibling.querySelector("input"); }); if (hasSiblingWithInput) { break; // current là container cao nhất cần dùng } current = current.parentElement; } if (!current) return ""; const labelTexts = Array.from(current.querySelectorAll("label")) .map((node) => node.textContent?.trim()) .filter((text) => text && text.length > 0); const otherTexts = Array.from(current.querySelectorAll("span, p, th, div")) .map((node) => node.textContent?.trim()) .filter((text) => text && text.length > 0); const allTexts = [...labelTexts, ...otherTexts]; return allTexts.join(" ").trim(); }
