import requests import re import pandas as pd from bs4 import BeautifulSoup # URL của request url = "https://dan.com/test/123" # Dữ liệu POST data = { 'even': '122', 'id': '1eihdkd' } # Headers, bao gồm Cookie headers = { 'Cookie': 'aws=123; asp=2222', 'Host': 'dan.com' } # Thực hiện request POST response = requests.post(url, data=data, headers=headers) # Kiểm tra nếu request thành công if response.status_code == 200: print("Request thành công, xử lý response...") # Nội dung response response_text = response.text # Sử dụng BeautifulSoup để phân tích cú pháp HTML soup = BeautifulSoup(response_text, 'html.parser') # Tìm tất cả các thẻ có href chứa "test.asp?id=" a_tags = soup.find_all('a', href=re.compile(r'test\.asp\?id=([^"]+)')) # Danh sách để lưu kết quả results = [] # Duyệt qua tất cả các thẻ tìm được for a_tag in a_tags: # Lấy id từ href của thẻ href = a_tag['href'] id_match = re.search(r'test\.asp\?id=([^"]+)', href) if id_match: id_value = id_match.group(1) # Tìm thẻ có class="test123" gần nhất trước thẻ td_tag = a_tag.find_previous('td', class_='test123') if td_tag: # Lấy text của thẻ td_text = td_tag.get_text(strip=True) else: td_text = 'Không tìm thấy thẻ td phù hợp' # Lưu kết quả results.append((id_value, td_text)) # Nếu tìm thấy kết quả, ghi vào file Excel if results: print(f"Tìm thấy {len(results)} cặp ID và text của thẻ . Ghi vào file Excel...") # Tạo DataFrame từ danh sách kết quả df = pd.DataFrame(results, columns=['ID', 'TD Text']) df.to_excel("id_td_text.xlsx", index=False) print("Hoàn thành! File 'id_td_text.xlsx' đã được tạo.") else: print("Không tìm thấy kết quả phù hợp.") else: print(f"Request thất bại với mã lỗi: {response.status_code}")
