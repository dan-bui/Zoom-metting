def parse_request_body(body): # Tách các cặp key=value từ request body và trả về dictionary params = {} if body: pairs = body.split('&') for pair in pairs: if '=' in pair: key, value = pair.split('=', 1) params[key] = value return params def merge_request_bodies(body1, body2): # Trích xuất tham số từ cả hai body params1 = parse_request_body(body1) params2 = parse_request_body(body2) # Tạo dictionary để lưu tất cả tham số và giá trị dưới dạng mảng all_params = {} # Xử lý body 1 for key, value in params1.items(): all_params[key] = [value] # Xử lý body 2, thêm giá trị vào mảng nếu tham số đã tồn tại for key, value in params2.items(): if key in all_params: if value not in all_params[key]: # Chỉ thêm nếu giá trị chưa có all_params[key].append(value) else: all_params[key] = [value] # In ra mảng nếu tham số có nhiều giá trị for key, values in all_params.items(): if len(values) > 1: print(f"{key}{values}") # Tạo request body chung, lấy giá trị đầu tiên trong mảng merged_body = [] for key, values in all_params.items(): merged_body.append(f"{key}={values[0]}") # Lấy giá trị đầu tiên return '&'.join(merged_body) # Test case 1 print("Test case 1:") body1 = "a=1&b=2&e=4" body2 = "a=1&b=2&c=3" result = merge_request_bodies(body1, body2) print(f"Request body chung: {result}") # Test case 2 print("\nTest case 2:") body1 = "a=5&b=2&e=4" body2 = "a=3&b=2&c=3" result = merge_request_bodies(body1, body2) print(f"Request body chung: {result}")
